/** * MWSS RO HR System — Google Apps Script Email Relay * ───────────────────────────────────────────────────── * Handles GET requests with URL parameters (no-cors browser calls). * Supports plain text AND HTML emails. * * PARAMETERS (via URL query string): * to — recipient email * subject — email subject * body — plain text body (fallback) * htmlBody — HTML body (optional — used for close-ticket email with button) * cc — comma-separated CC addresses * fromName — display name for the sender * * SETUP: * 1. script.google.com → paste this file * 2. Deploy → New Deployment → Web App * Execute as: Me | Who has access: Anyone * 3. Authorize → copy Web App URL * * IMPORTANT: Always create a NEW deployment after editing code. * "Edit existing deployment" does NOT pick up code changes. */ function doGet(e) { try { const to = e.parameter.to || ''; const subject = e.parameter.subject || ''; const body = e.parameter.body || ''; const htmlBody = e.parameter.htmlBody || ''; const ccRaw = e.parameter.cc || ''; const fromName = e.parameter.fromName || 'MWSS RO Personnel ALA'; if (!to || !subject) { return respond({ error: 'Missing required fields: to, subject' }); } // Build CC list const ccList = ccRaw ? ccRaw.split(',').map(function(s){ return s.trim(); }).filter(function(s){ return s; }) : []; const options = { name: fromName }; if (ccList.length) options.cc = ccList.join(','); // Use HTML body if provided, plain text as fallback if (htmlBody) { options.htmlBody = htmlBody; } GmailApp.sendEmail(to, subject, body || '(see HTML content)', options); Logger.log('Email sent | To: ' + to + ' | Subject: ' + subject + ' | HTML: ' + (htmlBody ? 'yes' : 'no')); return respond({ success: true, to: to }); } catch(err) { Logger.log('Email error: ' + err.message); return respond({ error: err.message }); } } function doPost(e) { // Support POST as well (for future extensibility) try { var data = {}; try { data = JSON.parse(e.postData.contents); } catch(_) {} const to = data.to || ''; const subject = data.subject || ''; const body = data.body || ''; const htmlBody = data.htmlBody || ''; const ccRaw = data.cc || ''; const fromName = data.fromName || 'MWSS RO Personnel ALA'; if (!to || !subject) return respond({ error: 'Missing fields' }); const ccList = ccRaw ? ccRaw.split(',').map(function(s){ return s.trim(); }).filter(function(s){ return s; }) : []; const options = { name: fromName }; if (ccList.length) options.cc = ccList.join(','); if (htmlBody) options.htmlBody = htmlBody; GmailApp.sendEmail(to, subject, body || '(see HTML content)', options); Logger.log('Email sent (POST) | To: ' + to); return respond({ success: true }); } catch(err) { return respond({ error: err.message }); } } function respond(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
